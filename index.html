# calculate mean and confidence intervals based off t-distribution
aspen_sugars <- aspen %>%
  group_by(phenology) %>%
  summarise(mean = mean(all_sugars, na.rm=TRUE),
            n = sum(!is.na(all_sugars)),
            SE = SE(all_sugars),
            lower95 = mean - qt(0.975, n-1)*SE,
            upper95 = mean + qt(0.975, n-1)*SE) %>% as_tibble()
aspen_sugars$phenology <- factor(aspen_sugars$phenology, levels = c("LEAF", "FULL", "SENE"))
# plot
p1.2.1 <- ggplot(aspen_sugars, aes(x=phenology, y = mean, pattern=phenology)) + 
   geom_col_pattern(fill='white',colour='black',
    pattern_density = 0.2,
    pattern_spacing = 0.04,
    pattern_key_scale_factor = 1) + 
  geom_errorbar(aes(ymin=lower95,ymax=upper95), width = 0.1) +
  geom_text(label=c("a","a","b"), hjust = -0.5, vjust=-0.5) +
  theme_classic() +
  theme(plot.title = element_text(hjust=0.5),
        plot.subtitle = element_text(hjust=0.5),
        legend.position = c(0.85, 0.8)) +
  labs(title = "Mean Total Sugar Content in Aspen Roots by Leaf State",
       subtitle = "DRAFT plot 1.2.1",
       x = "Tree Phenology",
       y = "Mean Total Sugars (ug/mg)",
       pattern = c("Phenological State")) +
  scale_pattern_manual(values=c("none","stripe","circle"),
                       labels = c("Leaf Out", "Full Expansion", "Senesced"))
